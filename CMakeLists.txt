cmake_minimum_required(VERSION 3.16)
project(productivity_protector LANGUAGES CXX)

set(CMAKE_AUTOMOC ON)

if(NOT DEFINED INSTALL_EXAMPLESDIR)
    set(INSTALL_EXAMPLESDIR "examples")
endif()

set(INSTALL_EXAMPLEDIR "${INSTALL_EXAMPLESDIR}/widgets/desktop/systray")

find_package(Qt6 REQUIRED COMPONENTS Core Gui Widgets)



if (WIN32)
#! [appicon_windows]
    set(app_icon_resource_windows "${CMAKE_CURRENT_SOURCE_DIR}/resources/proresource.rc")
    qt_add_executable(productivity_protector main.cpp window.cpp window.h ${app_icon_resource_windows})
#! [appicon_windows]
elseif (APPLE)
#! [appicon_macOS]
    # The MACOSX_BUNDLE_ICON_FILE variable is added to the Info.plist
    # generated by CMake. This variable contains the .icns file name,
    # without the path.
    set(MACOSX_BUNDLE_ICON_FILE photosurface.icns)

    # And the following tells CMake where to find and install the file itself.
    set(app_icon_macos "${CMAKE_CURRENT_SOURCE_DIR}/resources/photosurface.icns")
    set_source_files_properties(${app_icon_macos} PROPERTIES
           MACOSX_PACKAGE_LOCATION "Resources")

    qt_add_executable(productivity_protector MACOSX_BUNDLE main.cpp window.cpp window.h ${app_icon_macos})
#! [appicon_macOS]
else()
    qt_add_executable(productivity_protector main.cpp window.cpp window.h)
endif()


#qt_add_executable(productivity_protector
#    main.cpp
#    window.cpp window.h
#)

set_target_properties(productivity_protector PROPERTIES
    WIN32_EXECUTABLE TRUE
    MACOSX_BUNDLE TRUE
)

target_link_libraries(productivity_protector PUBLIC
    Qt::Core
    Qt::Gui
    Qt::Widgets
)

# Resources:
set(productivity_protector_resource_files
    "images/productivity_protector.png"
    "images/productivity_protector_bigger.png"
)

qt_add_resources(productivity_protector "productivity_protector"
    PREFIX
        "/"
    FILES
        ${productivity_protector_resource_files}
)

install(TARGETS productivity_protector
    RUNTIME DESTINATION "${INSTALL_EXAMPLEDIR}"
    BUNDLE DESTINATION "${INSTALL_EXAMPLEDIR}"
    LIBRARY DESTINATION "${INSTALL_EXAMPLEDIR}"
)
